  load.dataset <- function( object, x ) {
    object$setX( x[,c( seq(1, ncol(x) - 1,1))] )
    object$setY( x[,c(ncol(x))] )
  }

Print

  print.svm <- function(x) {
    print(sprintf("SVM object with: library: %s, kernel: %s, preprocess: %s, C: %.1f, 
                   gamma: %.3f, coef0: %.3f, degree: %d",
                  x$getLibrary(),
                  x$getKernel(),
                  x$getPreprocess(),
                  x$getC(),
                  x$getGamma(),
                  x$getCoef0(),
                  x$getDegree() ))
  }

  > print(model)
